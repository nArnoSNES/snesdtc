\textcolor{keywordtype}{int} main(\textcolor{keywordtype}{void}) \{
    u16 pad0;
    u8 sz=0,sx=0,angle=0;

    \textcolor{keywordflow}{while}(1) \{
        \textcolor{comment}{// Refresh pad values}
        \hyperlink{a00350_a42632e79292dc47a567aab4dc3e4407d}{scanPads}();

        \textcolor{keywordflow}{if} (pad0) \{
            \textcolor{comment}{// Update scrolling with current pad}

                \textcolor{keywordflow}{case} \hyperlink{a00350_aa27cad8fa018a58930b6622783a83072aa7b3a3d0503a46bec51c6e3da95747cc}{KEY\_A} : angle++; \hyperlink{a00368_a4d332cdc88b26fc9ec2cdd4f6710ce8c}{setMode7Rot}(angle); \textcolor{keywordflow}{break};

                \textcolor{keywordflow}{case} \hyperlink{a00350_aa27cad8fa018a58930b6622783a83072a6504370d9c6391e1a9da6a1a529b089d}{KEY\_RIGHT} : sz++; \textcolor{keywordflow}{break};
                \textcolor{keywordflow}{case} \hyperlink{a00350_aa27cad8fa018a58930b6622783a83072a612120b69c7dfd46086db7aaebdbcf65}{KEY\_LEFT}   : \textcolor{keywordflow}{if} (sx) sx--; \textcolor{keywordflow}{break};
                \textcolor{keywordflow}{case} \hyperlink{a00350_aa27cad8fa018a58930b6622783a83072aa9cdac7967bf7d88fdb761138a2a3416}{KEY\_DOWN}   : \textcolor{keywordflow}{if} (sz) sz--; \textcolor{keywordflow}{break};
                \textcolor{keywordflow}{case} \hyperlink{a00350_aa27cad8fa018a58930b6622783a83072a0848a442d907968b211b97bc2bd88acd}{KEY\_UP}       : sz++; \textcolor{keywordflow}{break};
            \}
        \}
        \hyperlink{a00341_a815e2626bd26d0e936d507a363ecd723}{WaitForVBlank}();
    \}
    \textcolor{keywordflow}{return} 0;
\}
